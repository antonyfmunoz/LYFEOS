/* app.js */
const today = new Date().toISOString().split('T')[0];
document.getElementById("date").textContent = today;

let routineData = JSON.parse(localStorage.getItem("routineData")) || [];

function renderRoutine() {
  const list = document.getElementById("routine-list");
  const todayList = document.getElementById("today-routine");
  list.innerHTML = "";
  todayList.innerHTML = "";

  routineData.forEach((block, index) => {
    const li = document.createElement("li");
    li.innerHTML = `<strong>${block.time}</strong> | ${block.name} — ${block.tasks.join(", ")} <button onclick="deleteBlock(${index})">❌</button>`;
    list.appendChild(li);

    const tli = document.createElement("li");
    tli.innerHTML = `<strong>${block.time}</strong> | ${block.name}<ul>${block.tasks.map(task => `<li>${task}</li>`).join("")}</ul>`;
    todayList.appendChild(tli);
  });
}

function deleteBlock(index) {
  routineData.splice(index, 1);
  localStorage.setItem("routineData", JSON.stringify(routineData));
  renderRoutine();
}

document.getElementById("routine-form").addEventListener("submit", (e) => {
  e.preventDefault();
  const time = document.getElementById("timeBlock").value;
  const name = document.getElementById("blockName").value;
  const tasks = document.getElementById("blockTasks").value.split(",").map(t => t.trim());

  routineData.push({ time, name, tasks });
  localStorage.setItem("routineData", JSON.stringify(routineData));
  e.target.reset();
  renderRoutine();
});

renderRoutine();

document.getElementById("saveBtn").addEventListener("click", () => {
  const log = {
    date: today,
    mental: document.getElementById("mental").value,
    physical: document.getElementById("physical").value,
    emotional: document.getElementById("emotional").value,
    reflection: document.getElementById("reflection").value,
    thoughts: document.getElementById("thoughts").value,
    content: document.getElementById("content").value
  };

  localStorage.setItem(`dailyLog-${today}`, JSON.stringify(log));
  document.getElementById("saveStatus").innerText = "✅ Log Saved!";
});
