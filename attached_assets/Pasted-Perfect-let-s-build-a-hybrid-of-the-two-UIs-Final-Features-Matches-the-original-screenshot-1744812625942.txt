🧠 app.js — Click-to-Complete Logic + Save State

// Restore mission state on load
document.addEventListener("DOMContentLoaded", () => {
  const stored = JSON.parse(localStorage.getItem("completedMissions")) || [];
  stored.forEach(i => {
    const mission = document.querySelectorAll(".mission-block")[i];
    if (mission) mission.classList.add("completed");
  });
});

function toggleComplete(el, index) {
  el.classList.toggle("completed");

  // Save completed state
  const allMissions = document.querySelectorAll(".mission-block");
  const completed = [];
  allMissions.forEach((m, i) => {
    if (m.classList.contains("completed")) completed.push(i);
  });

  localStorage.setItem("completedMissions", JSON.stringify(completed));
}
