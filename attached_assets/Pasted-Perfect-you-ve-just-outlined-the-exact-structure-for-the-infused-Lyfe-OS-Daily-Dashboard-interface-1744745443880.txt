// Date + Time Logic
const today = new Date();
document.getElementById("todayDate").innerText = today.toLocaleDateString('en-US', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });

function updateTime() {
  const now = new Date();
  document.getElementById("currentTime").innerText = now.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
}
setInterval(updateTime, 1000);

// Save Log Button
document.getElementById("saveLog").addEventListener("click", () => {
  const log = {
    date: today.toISOString().split('T')[0],
    wakeTime: document.getElementById("wakeTime").value,
    sleepTime: document.getElementById("sleepTime").value,
    mental: document.getElementById("mental").value,
    physical: document.getElementById("physical").value,
    emotional: document.getElementById("emotional").value,
    gratitude: document.getElementById("gratitude").value,
    tomorrowGoals: document.getElementById("tomorrowGoals").value,
    annualGoals: document.getElementById("annualGoals").value,
    thoughts: document.getElementById("thoughts").value,
    content: document.getElementById("content").value,
    research: document.getElementById("research").value,
    todoIdeas: document.getElementById("todoIdeas").value
  };

  localStorage.setItem(`dailyLog-${log.date}`, JSON.stringify(log));
  document.getElementById("saveStatus").innerText = "✅ Log saved!";
});
